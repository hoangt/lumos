from lumos import settings
import logging

from .. import TechModelError, BaseTechModel
from . import homoTFET30nm as tfet_30nm
from . import homoTFET60nm as tfet_60nm

_logger = logging.getLogger('TFETTechModel')
if settings.DEBUG:
    _logger.setLevel(logging.DEBUG)
else:
    _logger.setLevel(logging.INFO)



    [docs]class TFETTechModel(BaseTechModel):
    def __init__(self, model_name):
        if model_name == 'homoTFET30nm':
            tech = tfet_30nm
        elif model_name == 'homoTFET60nm':
            tech = tfet_60nm
        else:
            raise TechModelError('Unknown model name: {0}'.format(model_name))

        self._freq_dict = tech.freq_dict
        self._dynamic_power_dict = tech.dynamic_power_dict
        self._static_power_dict = tech.static_power_dict
        self.vnom_dict = tech.vnom_dict
        self.area_scale = tech.area_scale
        self.dynamic_power_scale = tech.dp_scale
        self.static_power_scale = tech.sp_scale
        self.fnom_scale = tech.fnom_scale
        self.perf_scale = tech.perf_scale

        super(TFETTechModel, self).__init__(model_name)


    [docs]    def dynamic_power(self, tech, vdd_mv, **kwargs):
        try:
            vdd_scale = self._dynamic_power_dict[tech][vdd_mv] / \
                self._dynamic_power_dict[tech][self.vnom(tech)]
        except KeyError:
            raise TechModelError('No dynamic power for Vdd: {0}mV at {1}nm'.format(vdd_mv, tech))

        return vdd_scale

    [docs]    def vmin(self, tech, **kwargs):
        try:
            return self._vmin
        except AttributeError:
            try:
                self._vmin = min(self._freq_dict[tech].keys())
            except KeyError:
                raise TechModelError('No freq for tech node at {0}nm'.format(tech))
            return self._vmin
