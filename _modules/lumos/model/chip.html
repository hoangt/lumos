[docs]    def build_chip(self, flp):
        """ Build a chip from floorplan, in variusNTV format

        Args:
           flp (str):
              The path of the floorplan file, in variusNTV format.

        Returns:
           N/A

        """
        self.units.clear()

        with open(flp, 'rb') as csvfile:
            csvreader = csv.reader(csvfile, delimiter="\t")
            fieldnames = csvreader.next()[0].split(' ')
            field_idx = dict((name, idx) for (idx,name)
                              in enumerate(fieldnames))
            for row in csvreader:
                unit = Unit()
                try:
                    unit_type = row[field_idx['Type']].lower()
                    name = row[field_idx['Unit']]
                    width = float(row[field_idx['Width']])
                    height = float(row[field_idx['Height']])
                    x = float(row[field_idx['X']])
                    y = float(row[field_idx['Y']])

                    if unit_type == 'memory':
                        nlines = int(row[field_idx['NLines']])
                        linesz = int(row[field_idx['LineSz']])
                        unit.set_prop(type=unit_type,name=name, width=width, height=height,
                                       x=x, y=y, nlines=nlines, linesz=linesz)

                    else: # should be logic unit
                        unit.set_prop(type=unit_type, name=name, width=width,
                                       height=height, x=x, y=y)

                    self.units[unit.name]=unit
                except ValueError:
                        print 'Error in floorplan format'

[docs]    def add_tmap(self, tmap):

        with open(tmap, 'rb') as csvfile:
            csvreader = csv.reader(csvfile, delimiter='\t')
            header = csvreader.next()
            field_idx = dict((name, idx+1) for (idx, name) in enumerate(header))
            for row in csvreader:
                name = row[0]
                unit = self.units[name]
                try:
                    vdd = float(row[field_idx['Vdd']])
                    d_mu = float(row[field_idx['MU']])
                    d_sd = float(row[field_idx['SD']])
                except ValueError:
                    print "Error in timing map"

                unit.add_delay(vdd, d_mu, d_sd)

    def __init__(self, dp0=0.307, sp0=0.1058, vdd0=0.765, vth=0.3):
        self.dp0 = dp0 # assume 0.05 to the 45nm dynamic power
        self.sp0 = sp0 # assume 0.01 to the 45nm static power
        self.vdd0= vdd0
        self.vth = vth
        self.delay_list = dict()

        self.vdd = vdd0
        self.dp = self.dp0
        self.sp = self.sp0
        self.sp_slope = 3.5

[docs]    def add_delay(self, vdd, d_mu, d_sd):
        self.delay_list[int(vdd*1000)] = {'mu':d_mu, 'sd': d_sd}
        if abs(vdd - self.vdd0) < 0.0001:
            self.delay0 = d_mu
            self.freq0 = 1 / self.delay0



[docs]    def get_delay(self):
        return self.delay_list[int(self.vdd*1000)]



[docs]    def set_vdd(self, vdd):
        self.vdd = vdd
        vdd0 = self.vdd0

        self.delay = self.delay_list[int(vdd*1000)]['mu']

        self.freq = 1 / self.delay

        self.dp = self.dp0 * (vdd/vdd0)**2 * (self.freq/self.freq0)
        self.sp = self.sp0 * 10 ** (vdd * self.sp_slope) / 10 ** (vdd0 * self.sp_slope)
        self.power = self.dp + self.sp
